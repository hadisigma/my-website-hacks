Einfache JavaScript-Animation
// Einfache Animation mit JavaScript
const element = document.querySelector('.animated-element');

function animate() {
  let position = 0;
  const interval = setInterval(() => {
    position += 1;
    element.style.transform = `translateX(${position}px)`;
    
    if (position > 300) {
      clearInterval(interval);
    }
  }, 10);
}

animate();

Python Data Analyse
# Einfache Datenanalyse mit Pandas
import pandas as pd
import matplotlib.pyplot as plt

# Daten laden
df = pd.read_csv('data.csv')

# Daten untersuchen
print(df.head())
print(df.describe())

# Daten visualisieren
plt.figure(figsize=(10, 6))
df['value'].plot(kind='hist', bins=20)
plt.title('Verteilung der Werte')
plt.xlabel('Wert')
plt.ylabel('Häufigkeit')
plt.savefig('histogram.png')
plt.show()
